\section*{0. Marco teórico común (la llave de todo)}
\addcontentsline{toc}{section}{0. Marco teórico común}

\subsection*{0.1 Espacio empírico de probabilidad}
Una base de datos es la terna $(\Om,\mathcal P(\Om),P_n)$ donde:
\begin{itemize}[leftmargin=1.4em,itemsep=1pt]
\item $\Om=\{\omega_1,\dots,\omega_n\}$ es el \textbf{conjunto de etiquetas (labels)} = los
\textbf{individuos estadísticos} (identifican cada fila de forma única). La etiqueta \emph{identifica};
la variable \emph{describe}. No confundirlas.
\item $\mathcal P(\Om)$ es el $\sigma$-álgebra trivial (todas las partes): $\Om$ finito $\Rightarrow$ todo subconjunto es evento.
\item Probabilidad \textbf{uniforme} (Laplace): $p_\omega=1/n$, y $P_n(A)=|A|/n$.
\item Una \textbf{variable} es una función $X:\Om\to\mathrm{Im}(X)$. Cada columna es una variable.
\end{itemize}

\subsection*{0.2 Partición inducida por una variable}
Toda $X$ con $\mathrm{Im}(X)=\{x_1,\dots,x_r\}$ induce una \textbf{partición} de $\Om$ en sus
\textbf{fibras / bloques}: $A_i:=X^{-1}(\{x_i\})=\{\omega:X(\omega)=x_i\}$. Es partición porque los
bloques son \emph{no vacíos} ($x_i\in\mathrm{Im}X$), \emph{disjuntos} ($X$ es función) y \emph{cubren}
$\Om$. $\sg(X)$ designa el \textbf{subespacio de funciones constantes sobre cada bloque} ($\dim=$ nº de bloques).

\subsection*{0.3 $L^2$ como espacio de Hilbert}
$L^2(\Om,\mathcal P(\Om),P_n)$ con
\[
\langle U,V\rangle=\E[UV]=\sum_\omega U(\omega)V(\omega)P_n(\{\omega\})=\tfrac1n\sum_{i=1}^n U(\omega_i)V(\omega_i),
\qquad \|U\|^2=\E[U^2].
\]
Como $\Om$ es finito, toda función está en $L^2$ y el espacio es completo (Hilbert) $\Rightarrow$ vale el Teorema de la Proyección.

\subsection*{0.4 Esperanza condicional = proyección ortogonal}
\begin{keybox}
\[ E(Y\mid\sg(X))=P_{\sg(X)}Y=\text{proyección ortogonal de }Y\text{ sobre }\sg(X). \]
\end{keybox}
Es el único $\hat Y\in\sg(X)$ que minimiza $\|Y-\hat Y\|^2$ (mejor predicción en media cuadrática).
En cada individuo vale \textbf{el promedio de $Y$ sobre su bloque}:
\[ E(Y\mid\sg(X))(\omega)=\frac1{|B|}\sum_{\omega'\in B}Y(\omega'),\quad B=\text{bloque de }\omega. \]
\textbf{Teorema de la Proyección (hecho central):} el residuo $Z=Y-\hat Y$ es ortogonal a todo el
subespacio: $\langle Z,W\rangle=0$ para todo $W\in\sg(X)$.\\
\textbf{Dato clave:} la constante $\mathbf 1$ siempre está en $\sg(X)$; como $\E[Z]=\langle Z,\mathbf 1\rangle$,
la ortogonalidad da $\E[Z]=0$.

\subsection*{0.5 Tres consecuencias inmediatas}
\begin{enumerate}[leftmargin=1.6em,itemsep=2pt]
\item \textbf{Insesgamiento / esperanza total:} $\mathbf 1\in\sg(X)$, $Z\perp\sg(X)$
$\Rightarrow\E[Z]=0\Rightarrow \E[E(Y\mid\sg(X))]=\E[Y]$.
\item \textbf{Pitágoras (norma):} $\|Y\|^2=\|\hat Y\|^2+\|Z\|^2$.
\item \textbf{Descomposición de varianza (ANOVA):}
\[ \Var(Y)=\underbrace{\Var(E(Y\mid\sg(X)))}_{\text{entre bloques (SC}_\text{inter})}
+\underbrace{\E(\Var(Y\mid\sg(X)))}_{\text{dentro (SC}_\text{intra})}. \]
\end{enumerate}

\subsection*{0.6 Fórmulas de proyección útiles}
\begin{itemize}[leftmargin=1.4em,itemsep=1pt]
\item Sobre una recta generada por $v\neq0$: $P_{\sg(v)}Y=\dfrac{\langle Y,v\rangle}{\langle v,v\rangle}v$.
\item \textbf{Suma de proyecciones SOLO si hay ortogonalidad:} si $\mathcal M\perp\mathcal N$ entonces
$P_{\mathcal M\oplus\mathcal N}=P_{\mathcal M}+P_{\mathcal N}$. Si no, \textbf{falla} (ecuaciones normales).
\end{itemize}
\begin{keybox}\small
\textbf{Mapa mental.} Variable $\to$ partición $\to$ subespacio $\sg(X)$ $\to$ proyección $=$
esperanza condicional $=$ promedio por bloque. De ahí: residuo ortogonal, esperanza total, Pitágoras,
ANOVA, $R^2$. Toda la materia es una idea geométrica repetida.
\end{keybox}

\section*{2. Coherencia en una base de datos (falacia ecológica)}
\textbf{Planteo:} votos con folio único, opción $\in\{$cand.1, cand.2, nulo, blanco$\}$; cada folio en mesa $\to$ local $\to$ distrito.

\inc{Inciso 1 — Etiquetas.} Son \textbf{los folios} ($\Omega_N$, $p_\omega=1/N$). La \textbf{opción de voto
NO es etiqueta}: es una variable $V:\Omega_N\to\{\text{cand.1, cand.2, nulo, blanco}\}$. Confundir folio con
valor de variable es el error de fondo. Los folios deben ser únicos, exhaustivos y mutuamente excluyentes.

\inc{Inciso 2 — ¿Mesa/Local/Distrito son variables?} Sí: funciones bien definidas sobre los folios,
$M,L,D:\Omega_N\to\{\cdot\}$. Sus particiones son \textbf{jerárquicas/encajonadas} (cada mesa en un único
local, cada local en un único distrito), luego Mesa refina a Local que refina a Distrito:
\[ \sg(D)\subseteq\sg(L)\subseteq\sg(M)\subseteq L^2(\Omega_N). \]
Existe la función de paso mesa$\mapsto$local$\mapsto$distrito, no al revés (lo fino determina lo grueso).

\inc{Inciso 3 — Condiciones para la tasa de cesantía comunal $C$.} Para ser variable de la base,
$C:\Omega_N\to\R$ (cesantía de la comuna del folio) debe cumplir: \emph{(i) coherencia jerárquica:}
asignación unívoca folio$\to$comuna, $K:\Omega_N\to\{\text{comunas}\}$ compatible con la jerarquía;
\emph{(ii) medibilidad:} $C$ constante por comuna, i.e. $C\in\sg(K)$.

\textbf{El punto de fondo (falacia ecológica):} como $C\in\sg(K)$ es constante dentro de cada comuna,
solo aporta variabilidad \emph{entre} comunas. Toda regresión factoriza por $E(V\mid C)=E(V\mid\sg(K))$:
\[ \Var(V)=\underbrace{\Var(E(V\mid\sg(K)))}_{\text{entre comunas (ve }C)}
+\underbrace{\E(\Var(V\mid\sg(K)))}_{\text{dentro: invisible}}. \]
Atribuir el efecto agregado al individuo es la \textbf{falacia ecológica} (Robinson, 1950); puede
invertirse (\textbf{paradoja de Simpson}: a nivel comunal ``más cesantía $\to$ más voto'', pero dentro
de cada comuna lo contrario). Para el nivel individual hace falta una variable definida sobre el folio
con partición más fina que la comunal.

\section*{3. Modelos lineales encajonados (nested)}
\textbf{Modelos:} M1: $E(Y\mid X_1)=\alpha_0+\alpha_1X_1$; M2: $E(Y\mid X_1,X_2)=\alpha_0+\alpha_1X_1+\alpha_2X_2$.
\textbf{Marco:} $\Omega_3$, uniforme $1/3$, $\langle U,V\rangle=\tfrac13\sum U_iV_i$.

\inc{Ejemplo 1 — regresores ORTOGONALES (se cumple la aditividad).}
$\mathbf 1=(1,1,1)$, $X=(-1,0,1)$. Productos: $\langle\mathbf 1,X\rangle=0$, $\langle X,X\rangle=\tfrac23$,
$\langle Y,\mathbf 1\rangle=\bar y$, $\langle Y,X\rangle=\tfrac{y_3-y_1}{3}$. Como $\mathbf 1\perp X$,
$\{\mathbf 1,X\}$ es base ortogonal y
\[ E(Y\mid\sg(\mathbf 1,X))=\bar y\,\mathbf 1+\tfrac{y_3-y_1}{2}X=E(Y\mid\sg(\mathbf 1))+E(Y\mid\sg(X)).\ \checkmark \]
La igualdad se debe a la \textbf{ortogonalidad}: proyectar sobre el plano = sumar proyecciones sobre cada eje.

\inc{Ejemplo 2 — regresores NO ortogonales (falla).}
$\mathbf 1=(1,1,1)$, $Z=(1,0,1)$, $\langle\mathbf 1,Z\rangle=\tfrac23\neq0$. Por \textbf{ecuaciones normales}
$\big(a\langle\mathbf 1,\mathbf 1\rangle+b\langle Z,\mathbf 1\rangle=\langle Y,\mathbf 1\rangle$, etc.$\big)$:
\[ a=y_2,\quad b=\tfrac{y_1+y_3}{2}-y_2,\quad E(Y\mid\sg(\mathbf 1,Z))=\Big(\tfrac{y_1+y_3}{2},y_2,\tfrac{y_1+y_3}{2}\Big). \]
Comparando en $\omega_2$: $y_2$ vs $\tfrac{y_1+y_2+y_3}{3}$; coinciden solo si $2y_2=y_1+y_3$. Al solaparse
$\sg(\mathbf 1)$ y $\sg(Z)$, sumar proyecciones cuenta dos veces la parte común.

\inc{Pregunta — ¿cuándo son ``encajonados''?}
\begin{enumerate}[leftmargin=1.6em,itemsep=2pt]
\item \textbf{Inclusión de subespacios:} $\sg(\mathbf 1)\subseteq\sg(\mathbf 1,X_1)\subseteq\sg(\mathbf 1,X_1,X_2)$
\emph{siempre}. Solo valida comparar ajustes (M2 nunca ajusta peor).
\item \textbf{Descomposición aditiva:} se cumple \emph{si y solo si} $X_2\perp\sg(\mathbf 1,X_1)$.
\end{enumerate}
\begin{keybox}
\textbf{Criterio:} M1 y M2 son ``encajonados'' en sentido fuerte solo cuando $X_2\perp\sg(\mathbf 1,X_1)$
(el coef.\ de $X_1$ no cambia al agregar $X_2$). Si los predictores están correlacionados, agregar $X_2$
altera el coef.\ de $X_1$. La forma correcta es \textbf{ortogonalizar primero} (\textbf{Frisch--Waugh--Lovell}).
La receta ingenua ``ajusto, agrego, comparo'' solo es interpretable bajo ortogonalidad.
\end{keybox}

\section*{4. Teorema de Pitágoras ($R^2$ vs $r_n^2$)}
\textbf{Setup:} $Y=\hat Y+Z$, $\hat Y=E(Y\mid\sg(X))$, $Z=Y-\hat Y$, (1) $\|Y\|^2=\|\hat Y\|^2+\|Z\|^2$.
Recordatorio: $\mathbf 1\in\sg(X)\Rightarrow\langle Z,\mathbf 1\rangle=\E[Z]$.

\inc{Inciso 1 — $\E(Y-E(Y\mid\sg(X)))=0$.} $\mathbf 1\in\sg(X)$ y $Z\perp\sg(X)$
$\Rightarrow\langle Z,\mathbf 1\rangle=\E[Z]=0$. Consecuencia: $\E[\hat Y]=\E[Y]=:\mu$.

\inc{Inciso 2 — equivalencia (1)$\Leftrightarrow$(2).} Con $\Var(U)=\E[U^2]-[\E U]^2$ y $\E[\hat Y]=\E[Y]=\mu$,
$\E[Z]=0$. Restando $\mu^2$ a $\E[Y^2]=\E[\hat Y^2]+\E[Z^2]$ y agrupando $-\mu^2$ con el término de $\hat Y$:
\[ \underbrace{\E[Y^2]-\mu^2}_{\Var(Y)}=\underbrace{(\E[\hat Y^2]-\mu^2)}_{\Var(\hat Y)}+\underbrace{\E[Z^2]}_{\Var(Z)}. \]
``Restar $\mu^2$'' es reversible $\Rightarrow$ (1) y (2) equivalentes (descansa en el inciso 1).

\inc{Inciso 3 — comparar $R^2$ y $r_n^2$.} Con $a=\|\hat Y\|^2=\E[\hat Y^2]$, $b=\|Y\|^2=\E[Y^2]$:
\[ r_n^2=\frac{\|E(Y\mid\sg(X))\|^2}{\|Y\|^2}=\frac{a}{b}\ \text{(crudos)},\qquad
R^2=\frac{\Var(\hat Y)}{\Var(Y)}=\frac{a-\mu^2}{b-\mu^2}\ \text{(centrados)}. \]
Orden: $b=a+\E[Z^2]\Rightarrow a\le b$; por Jensen $a\ge\mu^2$. Producto cruzado ($R^2\le r_n^2\iff\mu^2 a\le\mu^2 b$):
\begin{keybox}\[ R^2\le r_n^2. \]\end{keybox}
\textbf{Igualdad} sii $\mu=\E[Y]=0$ (centrada) o $a=b$ ($Y=\hat Y$ c.s.). $R^2$ es el genuino coeficiente
de determinación (invariante a traslaciones); $r_n^2$ usa momentos crudos y \textbf{sobrestima}.

\textbf{Ejemplo:} $\Omega=\{1,2,3,4\}$ uniforme, bloques $\{1,2\},\{3,4\}$, $Y=(0,2,2,4)$, $\hat Y=(1,1,3,3)$,
$\mu=2$, $b=6$, $a=5$, $\Var(Y)=2$, $\Var(\hat Y)=1$, $Z=(-1,1,-1,1)$. Entonces
$r_n^2=\tfrac56\approx0.833$, $R^2=\tfrac12=0.5$ ($R^2<r_n^2$).

\section*{5. Esperanza condicional (construcción y ley total)}
\textbf{Setup:} $H$ Hilbert, $M$ cerrado; dato: para cada $x$ existe único $y_0\in M$ con $\|x-y_0\|$ mínimo.

\inc{Inciso 1 — $P:H\to M$ es proyección ortogonal.}
\begin{itemize}[leftmargin=1.4em,itemsep=1pt]
\item \textbf{Definición:} $Px:=y_0$ (bien definido por unicidad).
\item \textbf{Lema clave:} $y_0$ minimiza $\iff(x-y_0)\perp M$. ($\Leftarrow$) Pitágoras
$\|x-y\|^2=\|x-y_0\|^2+\|y_0-y\|^2$; ($\Rightarrow$) $\varphi(t)=\|x-(y_0+tm)\|^2$ tiene mínimo en
$t=0\Rightarrow\langle x-y_0,m\rangle=0$.
\item \textbf{Lineal} (por unicidad), \textbf{idempotente} ($P^2=P$), \textbf{autoadjunto}
($\langle Px,z\rangle=\langle x,Pz\rangle$), \textbf{contracción} ($\|P\|\le1$), $H=M\oplus M^\perp$.
\end{itemize}

\inc{Inciso 2 — la proyección ES la esperanza condicional.} En $L^2(\Omega,\mathcal F,P)$ con
$\langle X,Y\rangle=\E[XY]$, $M=L^2(\Omega,\mathcal G,P)$ es cerrado. La definición de $E[X\mid\mathcal G]$
($\mathcal G$-medible con $\E[(X-Z)W]=0\ \forall W\in M$) es exactamente la condición de ortogonalidad
$X-Z\perp M$ $\Rightarrow E[X\mid\mathcal G]=PX$ (mejor predicción en media cuadrática).

\inc{Inciso 3 — $\E[E(Y\mid\sg(X))]=\E(Y)$.} Concretando en $\Om$ finito, $M=\sg(X)$, $\mathbf 1\in M$ y
$\E[Z]=\langle Z,\mathbf 1\rangle$. Como $Y-\hat Y\perp M$:
\[ \E[Y]-\E[\hat Y]=\langle Y-\hat Y,\mathbf 1\rangle=0\Rightarrow\E[\hat Y]=\E[Y]. \]
Verificación: $\hat Y|_{B_j}=\E[Y\mid B_j]$ y $\E[\hat Y]=\sum_j P(B_j)\E[Y\mid B_j]=\E[Y]$ (ley de la esperanza total).

\section*{6. Igualdad del paralelogramo}
\textbf{Enunciado:} en un espacio con producto interno, $\|x+y\|^2+\|x-y\|^2=2\|x\|^2+2\|y\|^2$.

\inc{Inciso 1 — Demostración.} Con $\|v\|^2=\langle v,v\rangle$ (único uso del producto interno),
bilinealidad y simetría: $\|x\pm y\|^2=\|x\|^2\pm2\langle x,y\rangle+\|y\|^2$. Al sumar, los cruzados
$\pm2\langle x,y\rangle$ \textbf{se cancelan}. (Complejo: cruzados $2\,\mathrm{Re}\langle x,y\rangle$, cancela igual.)

\inc{Inciso 2 — Relevancia para la esperanza condicional.} Es una \textbf{proyección ortogonal en $L^2$},
cuya existencia/unicidad (Teorema de la Proyección) usa la igualdad: para una sucesión minimizante $(Y_n)$
con $\|X-Y_n\|\to d$, aplicando a $a=X-Y_n$, $b=X-Y_m$,
\[ \|Y_m-Y_n\|^2=2\|X-Y_n\|^2+2\|X-Y_m\|^2-4\Big\|X-\tfrac{Y_n+Y_m}{2}\Big\|^2\le 2\|X-Y_n\|^2+2\|X-Y_m\|^2-4d^2\to0, \]
luego $(Y_n)$ es de Cauchy; por completitud converge a $Y^*\in M$. Sin la igualdad (norma sin producto
interno) el argumento se rompe.

\inc{Inciso 3 — $\|\cdot\|_\infty$ en $\R^n$ ($n\ge2$).} $x=e_1$, $y=e_2$: $\|x\|_\infty=\|y\|_\infty=1$,
$\|x\pm y\|_\infty=1$. LI$=1+1=2$, LD$=2+2=4$. $2\neq4\Rightarrow$ falla.

\inc{Inciso 4 — $\|\cdot\|_\infty$ en $C[0,1]$.} $F(x)=1$, $G(x)=x$: $\|F\|_\infty=\|G\|_\infty=1$,
$\|F+G\|_\infty=2$, $\|F-G\|_\infty=1$. LI$=4+1=5$, LD$=4$. $5\neq4\Rightarrow$ falla (KS usa $\|\cdot\|_\infty$,
que no induce geometría de Hilbert).

\section*{7. El término de error — dos miradas}
\textbf{Modelo:} $y=X\beta+\varepsilon$, $\mathrm{rango}(X)=p$. Ambas llegan al mismo
$\hat\beta=(X^\top X)^{-1}X^\top y$ por caminos opuestos: el mismo objeto $\varepsilon$ con dos significados.

\textbf{Mirada de Gauss (probabilística).} $\varepsilon$ es objeto \textbf{primitivo} (v.a.\ con ley supuesta):
$\E[\varepsilon]=0$, $\Var(\varepsilon)=\sigma^2 I_n$ (homocedasticidad + no correlación); para inferir
$\varepsilon\sim\mathcal N(0,\sigma^2I_n)$. \emph{Argumento histórico:} la media aritmética como estimador
plausible $\Rightarrow$ la ley que la hace máximo-verosímil es la normal $\Rightarrow$ MV $=$ minimizar
$\sum\varepsilon_i^2$. Sin normalidad (Gauss--Markov): $\hat\beta$ es \textbf{BLUE}. Estimación:
$\partial_\beta\|y-X\beta\|^2=0\Rightarrow X^\top X\hat\beta=X^\top y$. Estilo: paramétrico, inferencial
($\Var(\hat\beta)=\sigma^2(X^\top X)^{-1}$, tests $t$,$F$, IC). Mentalidad: de la probabilidad a las propiedades.

\textbf{Mirada de la Proyección (geométrica).} $\R^n$ con $\langle u,v\rangle=u^\top v$, $\mathcal C(X)=\{X\beta\}$.
$y=\hat y+e$, $\hat y\in\mathcal C(X)$, $e\perp\mathcal C(X)$: el error \textbf{se define} como complemento
ortogonal. Ecuaciones normales $=$ ortogonalidad: $X^\top e=0$. $P=X(X^\top X)^{-1}X^\top$, $\hat y=Py$,
$e=(I-P)y$. Pitágoras $=$ ANOVA sin distribución: $\|y\|^2=\|\hat y\|^2+\|e\|^2$. En $L^2$, la no correlación
del error con los regresores es \textbf{teorema} (proyección), no supuesto. Estilo: geométrico, libre de distribución.

\begin{center}
\begin{tabular}{@{}p{4.0cm}p{4.4cm}p{4.4cm}@{}}
\toprule
\textbf{Criterio} & \textbf{Gauss} & \textbf{Proyección}\\
\midrule
Naturaleza del error & Primitivo (v.a.\ con ley) & Derivado (residuo ortogonal)\\
No correlación & Supuesto & Consecuencia automática\\
Ecuaciones normales & Anular $\partial S/\partial\beta$ & Traducir $X^\top e=0$\\
Por qué MCO & Verosimilitud normal / BLUE & Proyección $=$ punto más cercano\\
Distribución & Central desde el inicio & Postergada (solo para inferir)\\
Estilo & Inferencial/paramétrico & Geométrico/estructural\\
\bottomrule
\end{tabular}
\end{center}
\textbf{Síntesis:} Gauss responde \emph{por qué es óptimo} (inferencial); la proyección, \emph{por qué tiene esa
forma} (estructural). Distinción fina: errores verdaderos $\varepsilon$ $\neq$ residuos $e=(I-P)\varepsilon$.

\section*{8. Independencia de funciones ($n=13$ primo)}
\textbf{Setup:} $\Om$ empírico uniforme, $X:\Omega_{13}\to\{x_1,\dots,x_r\}$, $Y:\Omega_{13}\to\{y_1,\dots,y_q\}$.

\inc{Inciso 1 — ¿$r$ arbitrario?} No: $1\le r\le n$, pues $n=\sum_{i=1}^r|A_i|\ge r$ (bloques no vacíos) y
$r\ge1$. $r=1\iff X$ constante; $r=n\iff X$ inyectiva. Con $n=13$: $r\in\{1,\dots,13\}$.

\inc{Inciso 2 — Partición de $(X,Y)$.} La fibra es $(X,Y)^{-1}(\{(x_i,y_j)\})=A_i\cap B_j$, luego la partición
es $\{A_i\cap B_j:A_i\cap B_j\neq\varnothing\}$: el \textbf{refinamiento común} de $\{A_i\}$ y $\{B_j\}$
(pues $A_i=\bigsqcup_j(A_i\cap B_j)$; nº de bloques $\le r\cdot q$ y $\le n$).

\inc{Inciso 3 — ¿Qué $Y$ son independientes de $X$ fija?} Independencia $\iff
P(X=x_i,Y=y_j)=P(X=x_i)P(Y=y_j)$. En el espacio empírico:
\[ \frac{|A_i\cap B_j|}{13}=\frac{|A_i|}{13}\cdot\frac{|B_j|}{13}\iff |A_i\cap B_j|=\frac{|A_i||B_j|}{13}. \]
El lado izquierdo es entero $\Rightarrow 13\mid|A_i||B_j|$. Como \textbf{13 es primo} (Euclides):
$13\mid|A_i|$ o $13\mid|B_j|$; y como $1\le|A_i|,|B_j|\le13$, esto fuerza $|A_i|=13$ ($X$ constante) o
$|B_j|=13$ ($Y$ constante).
\begin{itemize}[leftmargin=1.4em,itemsep=1pt]
\item \textbf{$X$ no constante ($r\ge2$):} $\Rightarrow$ $Y$ debe ser constante ($q=1$).
\item \textbf{$X$ constante:} toda $Y$ es independiente de $X$.
\end{itemize}
\begin{keybox}
\textbf{Papel de la primalidad de 13:} la independencia exige $|A_i\cap B_j|=|A_i||B_j|/n$ entero
(``rectángulo''); dos variables no triviales independientes requerirían $n=a\cdot b$ con $1<a,b<n$,
imposible si $n$ es primo. Por eso sobre $\Omega_{13}$ dos variables solo pueden ser independientes si al
menos una es constante. Con $n$ compuesto (p.\,ej.\ $n=12$) sí existirían no constantes genuinamente independientes.
\end{keybox}

\section*{Apéndice — Conceptos transversales}
\begin{itemize}[leftmargin=1.4em,itemsep=2pt]
\item \textbf{Etiqueta vs.\ variable:} la etiqueta identifica; la variable describe (preguntas 1, 2, 8).
\item \textbf{Partición / $\sg(X)$:} fibras de la variable; $\sg(X)$ = funciones constantes por bloque.
\item \textbf{Esperanza condicional = proyección ortogonal} = promedio por bloque = mejor predictor.
\item \textbf{Teorema de la Proyección} $\Rightarrow$ residuo ortogonal $\Rightarrow$ esperanza total, Pitágoras, ANOVA.
\item \textbf{$\mathbf 1\in\sg(X)$ siempre} $\Rightarrow$ $\E[Z]=0$ (insesgamiento).
\item \textbf{Ortogonalidad lo es todo:} proyecciones se suman solo si los subespacios son ortogonales
(p.\,3); no correlación $\equiv$ ortogonalidad en $L^2$ centrado (p.\,7).
\item \textbf{$R^2$ (centrado) $\le r_n^2$ (crudo)}; iguales si $\mu=0$ o ajuste perfecto (p.\,4).
\item \textbf{Norma del supremo} no viene de producto interno (falla paralelogramo) $\Rightarrow$ no sirve
para definir esperanza condicional (p.\,6).
\item \textbf{Primalidad y rectángulos:} $n$ primo $\Rightarrow$ independencia solo con variable constante (p.\,8).
\end{itemize}
